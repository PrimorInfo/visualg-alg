Algoritmo "DEPENDENTES"

Var
   nome: caracter
   sal, nsal: real
   depen: inteiro

Inicio
   escreva("Qual é o nome do funcionário? ")
   leia(nome)
   escreva("Qual é o salário do funcionário? R$")
   leia(sal)
   escreva("Qual é a qtde de dependentes? ")
   leia(depen)
   escolha(depen)
          caso 0
           nsal <- sal + (sal*5/100)
          caso 1, 2, 3
           nsal <- sal + (sal*10/100)
          caso 4, 5, 6
           nsal <- sal + (sal*15/100)
          outroCaso
           nsal <- sal + (sal*18/100)
    fimescolha
    escreva("O novo salário de  ", nome, " será de: R$", nsal:4:2)

Fimalgoritmo
